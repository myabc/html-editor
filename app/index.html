Substance HTML Editor

HTML Editor is a fully customizable and highly reliable and performant Rich Text Editing Component for your web app. Unlike most web-based editors, Substance operates on a Javascript data model, and uses HTML just as an input and output format. With Substance you can build full-fledged editing applications..

Usage

This editor depends on React and jQuery.


var React = require("react");
var HtmlEditor = require("substance-html-editor");
var $$ = React.createElement;

React.render(
  $$(HtmlEditor, {
    content: htmlContent,
    onContentChanged: function(html) {
      console.log('Edited content');
    }
  }),
  document.getElementById('editor_container')
);

You can include markup tools you want to use, and customize their markup by passing children elements in the classic React style.


var React = require("react");
var HtmlEditor = require("substance-html-editor");
var $$ = React.createElement;

React.render(
  $$(HtmlEditor, {
    content: htmlContent,
    tools: [
      $$(ToolComponent, { tool: 'emphasis', title: 'Emphasis', classNames: ['button', 'tool']}, "Emphasis"),
      $$(ToolComponent, { tool: 'strong', title: 'Strong', classNames: ['button', 'tool']}, "Strong"),
    ],
    onContentChanged: function(html) {
      console.log('Edited content');
    }
  }),
  document.getElementById('editor_container')
);
